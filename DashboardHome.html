import React, { useEffect, useState } from 'react';

import ClientSummaryCards from '../../components/dashboard/ClientSummaryCards';

import { supabase } from '../../services/supabaseClient';



const DashboardHome = () => {

const [stats, setStats] = useState({ balance: 0, activeOrders: 0, appointments: 0 });



useEffect(() => {

// Fetch user data from Supabase

const fetchData = async () => {

const { data } = await supabase

.from('profiles')

.select('wallet_balance, active_orders_count, appointments_count')

.single();

if (data) setStats({

balance: data.wallet_balance,

activeOrders: data.active_orders_count,

appointments: data.appointments_count

});

};

fetchData();

}, []);



return (

<div className="dashboard-page">

<h1>Welcome back, Client</h1>

<ClientSummaryCards data={stats} />

<section className="quick-actions">

<h2>Quick Actions</h2>

<button className="btn-primary">New Service Order</button>

<button className="btn-secondary">Book Appointment</button>

</section>

</div>

);

};



export default DashboardHome;
